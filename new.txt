# Lead Ace: AI-Powered Lead Generation Platform

Lead Ace is an advanced platform that streamlines the lead generation lifecycle using Generative AI. It allows sales teams to discover, capture, and qualify leads from virtually any source.

## Key Features

### 1. Intelligent Discovery
- **Natural Language Search**: Describe your target market in plain text.
- **AI-Driven Keywords**: The system optimizes search terms to find the most relevant businesses.

### 2. Versatile Lead Capture
- **Web & Bulk URL Capture**: Extract contact details from websites or partner lists.
- **Ad & Social Capture**: Upload screenshots of advertisements or social posts to identify leads.
- **Street View Prospecting**: Explore areas via Google Street View to identify and capture physical business leads with photographic evidence.

### 3. Data Intelligence
- **Automated Enrichment**: Fetch company data including tech stacks, revenue, and funding info.
- **Predictive Scoring**: Prioritize leads based on an AI-calculated quality score.
- **AI Summaries**: Get concise, one-paragraph briefings on every lead.

### 4. Management & Analytics
- **Lead Pipeline**: Track status from discovery to qualification.
- **Interactive Dashboard**: Visualize conversion rates and lead quality.
- **CSV Export**: Easily move your leads into other CRM tools.

## Tech Stack
- **Framework**: Next.js (App Router)
- **Database/Auth**: Firebase (Firestore & Authentication)
- **AI**: Google Genkit & Gemini 1.5 Flash
- **UI**: ShadCN UI, Tailwind CSS, Lucide Icons

'use server';

import { ai } from '@/ai/genkit';
import { z } from 'genkit';
import { logger } from '@/lib/logger';
import { ExtractedLeadDataSchema, type ExtractedLeadData } from '@/types';
import fetch from 'node-fetch';

const AnalyzeAdForLeadInputSchema = z.object({
  photoDataUri: z.string().describe("A screenshot of an ad as a data URI."),
});

const AnalyzeAdForLeadOutputSchema = z.object({
    extractedData: ExtractedLeadDataSchema.nullable().optional(),
    message: z.string().describe("A message describing the outcome."),
});
export type AnalyzeAdForLeadOutput = z.infer<typeof AnalyzeAdForLeadOutputSchema>;

const imageAnalysisPrompt = ai.definePrompt({
    name: 'analyzeAdImageForUrlPrompt',
    input: { schema: AnalyzeAdForLeadInputSchema },
    output: { schema: z.object({ companyName: z.string().optional(), websiteUrl: z.string().url().optional() }) },
    prompt: `You are an expert at analyzing advertisement screenshots (like Google or Facebook ads). Your most important task is to identify and extract the main website URL being advertised. Also, extract the company name.

Image to analyze: {{media url=photoDataUri}}`
});

const htmlAnalysisPrompt = ai.definePrompt({
    name: 'analyzeWebsiteForDetailsPrompt',
    input: { schema: z.object({ htmlContent: z.string(), companyName: z.string().optional() }) },
    output: { schema: ExtractedLeadDataSchema },
    prompt: `You are an expert at analyzing a company's website HTML to find contact information. Find the primary contact phone number, physical address, and a contact person's name and title if available. Use the provided company name for context.

Company Name: {{companyName}}

Website HTML:
\`\`\`html
{{htmlContent}}
\`\`\`
`
});

export async function analyzeAdForLead(
  input: z.infer<typeof AnalyzeAdForLeadInputSchema>
): Promise<AnalyzeAdForLeadOutput> {
  return analyzeAdForLeadFlow(input);
}

const analyzeAdForLeadFlow = ai.defineFlow(
  {
    name: 'analyzeAdForLeadFlow',
    inputSchema: AnalyzeAdForLeadInputSchema,
    outputSchema: AnalyzeAdForLeadOutputSchema,
  },
  async (input) => {
    logger.info('[Ad Analysis] Step 1: Analyzing image for URL.');
    const imageAnalysisResult = await imageAnalysisPrompt(input);
    const { companyName, websiteUrl } = imageAnalysisResult.output || {};

    if (!websiteUrl) {
      logger.warn('[Ad Analysis] Could not find a website URL in the image.');
      return { message: "The AI could not identify a website URL in the ad. Please try a different image or enter the lead manually." };
    }
    logger.info(`[Ad Analysis] Found URL: ${websiteUrl} and Company: ${companyName}`);

    let htmlContent = '';
    try {
        logger.info(`[Ad Analysis] Step 2: Fetching content from ${websiteUrl}`);
        const response = await fetch(websiteUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36' }});
        if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
        htmlContent = await response.text();
    } catch (error: any) {
        logger.error(`[Ad Analysis] Failed to fetch URL ${websiteUrl}: ${error.message}`);
        return { 
            extractedData: { companyName, website: websiteUrl },
            message: `AI found a website, but failed to load it. Please review the extracted data.` 
        };
    }

    logger.info('[Ad Analysis] Step 3: Analyzing HTML for contact details.');
    const htmlAnalysisResult = await htmlAnalysisPrompt({ htmlContent, companyName });
    const scrapedData = htmlAnalysisResult.output;
    
    const finalData: ExtractedLeadData = {
        name: scrapedData?.name,
        title: scrapedData?.title,
        companyName: scrapedData?.companyName || companyName,
        email: scrapedData?.email,
        phone: scrapedData?.phone,
        website: scrapedData?.website || websiteUrl,
        details: scrapedData?.details,
    };

    return {
        extractedData: finalData,
        message: "Successfully analyzed ad and website. Please verify the details below."
    };
  }
);

'use server';

import { ai } from '@/ai/genkit';
import { z } from 'genkit';
import { logger } from '@/lib/logger';
import { ExtractedLeadDataSchema } from '@/types';
import type { ExtractedLeadData } from '@/types';

const AnalyzeImageForLeadInputSchema = z.object({
  photoDataUri: z.string(),
});
export type AnalyzeImageForLeadInput = z.infer<typeof AnalyzeImageForLeadInputSchema>;

const AnalyzeImageForLeadOutputSchema = z.object({
    extractedData: ExtractedLeadDataSchema.optional(),
    message: z.string(),
});
export type AnalyzeImageForLeadOutput = z.infer<typeof AnalyzeImageForLeadOutputSchema>;

export async function analyzeImageForLead(
  input: AnalyzeImageForLeadInput
): Promise<AnalyzeImageForLeadOutput> {
  return analyzeImageForLeadFlow(input);
}

const prompt = ai.definePrompt({
  name: 'analyzeImageForLeadPrompt',
  input: { schema: AnalyzeImageForLeadInputSchema },
  output: { schema: ExtractedLeadDataSchema.optional() },
  prompt: `You are an expert at analyzing images of advertisements, business cards, and social media posts to extract lead information.
Analyze the provided image and extract: name, job title, company name, email, phone number, and website.
Image to analyze: {{media url=photoDataUri}}`,
});

const analyzeImageForLeadFlow = ai.defineFlow(
  {
    name: 'analyzeImageForLeadFlow',
    inputSchema: AnalyzeImageForLeadInputSchema,
    outputSchema: AnalyzeImageForLeadOutputSchema,
  },
  async (input) => {
    const { output } = await prompt(input);
    if (!output || (!output.name && !output.companyName)) {
      return { message: "The AI could not identify a potential lead in this image." };
    }
    return { 
        extractedData: output,
        message: "AI analysis complete. Please verify the extracted information below."
    };
  }
);

'use server';

import { z } from 'genkit';
import { enrichLeadData } from '@/lib/enrichment';
import { calculateLeadScore } from '@/lib/scoring';
import { createLead } from '@/lib/db';
import { logger } from '@/lib/logger';
import { NewLeadSchema } from '@/types';
import { ai } from '../genkit';

export const CreateLeadOutputSchema = z.object({
    leadId: z.string(),
    message: z.string()
});

export const createLeadFromForm = ai.defineFlow(
  {
    name: 'createLeadFromFormFlow',
    inputSchema: NewLeadSchema,
    outputSchema: CreateLeadOutputSchema,
  },
  async (leadData) => {
    const enrichedData = await enrichLeadData(leadData);
    const quality = calculateLeadScore(enrichedData as any);
    const finalLeadData = { ...enrichedData, quality, createdAt: Date.now() };
    const newLead = await createLead(finalLeadData as any);
    return { leadId: newLead.id, message: `Successfully created new lead: ${newLead.name}.` };
  }
);

'use server';

import { ai } from '@/ai/genkit';
import { z } from 'genkit';
import { logger } from '@/lib/logger';
import { ExtractedLeadDataSchema } from '@/types';
import fetch from 'node-fetch';

const ExtractLeadFromUrlInputSchema = z.object({
  url: z.string().url(),
});

const prompt = ai.definePrompt({
  name: 'extractLeadFromUrlPrompt',
  input: { schema: z.object({ htmlContent: z.string() }) },
  output: { schema: ExtractedLeadDataSchema.optional() },
  prompt: `Extract lead info from this HTML: {{htmlContent}}`,
});

export async function extractLeadFromUrl(input: z.infer<typeof ExtractLeadFromUrlInputSchema>) {
  const response = await fetch(input.url);
  const htmlContent = await response.text();
  const { output } = await prompt({ htmlContent });
  return { extractedData: output, message: "Analysis complete." };
}

--- FILENAME: src/ai/flows/find-leads.ts ---
'use server';

import { z } from 'genkit';
import { scrapeWithAI } from '@/lib/scraper';
import { enrichLeadData } from '@/lib/enrichment';
import { calculateLeadScore } from '@/lib/scoring';
import { suggestLeadSearchTerms } from './suggest-lead-search-terms';
import { createLead } from '@/lib/db';
import { logger } from '@/lib/logger';
import { ai } from '../genkit';

export const findLeads = ai.defineFlow(
  {
    name: 'findLeadsFlow',
    inputSchema: z.object({ searchContext: z.string() }),
    outputSchema: z.object({ totalLeadsFound: z.number(), message: z.string() }),
  },
  async (input) => {
    const result = await suggestLeadSearchTerms({ idealLeadProfileDescription: input.searchContext });
    const terms = result.searchTerms.split(',').map(t => t.trim());
    const scrapedLeads = await scrapeWithAI({ source: 'linkedin', searchTerms: terms, maxResults: 2 });
    
    for (const lead of scrapedLeads) {
      const enriched = await enrichLeadData(lead);
      await createLead({ ...enriched, status: 'New', quality: calculateLeadScore(enriched as any), source: 'AI Search' } as any);
    }
    return { totalLeadsFound: scrapedLeads.length, message: "Lead discovery complete." };
  }
);

import { collection, addDoc, getDocs, deleteDoc, doc, updateDoc, onSnapshot, query, where, getDoc } from 'firebase/firestore';
import { auth, db } from './firebase';

export async function createLead(data) {
  const userId = auth.currentUser?.uid;
  return addDoc(collection(db, 'leads'), { ...data, userId, createdAt: Date.now() });
}

export function listenToLeads(callback) {
  const userId = auth.currentUser?.uid;
  const q = query(collection(db, 'leads'), where("userId", "==", userId));
  return onSnapshot(q, (ss) => callback(snapshot.docs.map(d => ({ id: d.id, ...d.data() }))));
}
